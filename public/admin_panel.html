async function loadPayments() { const res = await fetch(`${backendUrl}/api/payments`); const data = await res.json(); const tbody = document.querySelector("#payments-table tbody"); tbody.innerHTML = ""; data.forEach(p => { const row = document.createElement("tr"); row.innerHTML = ` ${p.business_name} ${p.credits} ₹${p.amount} ₹${p.unit_price} ${p.razorpay_payment_id} ${p.timestamp} `; tbody.appendChild(row); }); }
